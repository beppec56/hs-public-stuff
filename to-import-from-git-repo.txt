git checkout -b new_dest_target master (optional step, the new branch will track the other git repo)

git remote add rack_remote git@github.com:schacon/Rack.git or git remote add /home/beppe...../Rack
git fetch rack_remote
git merge -s ours --no-commit rack_remote/master
git read-tree --prefix=Rack/ -u rack_remote/master
git commit -m "Imported Rack as a subtree."
